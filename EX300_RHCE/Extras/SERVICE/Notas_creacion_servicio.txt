[Unit]
	Description=
	Documentation=
	Requires=unidades que tienen que estar activas para que esta esté activa, si alguna falla ésta no se activa (no orden, se activan simultaneamente, mejor Wants)
	Wants=unidades que tienen que estar activas para que esta esté activa (no orden, se activan simultaneamente, pero si una falla no impacta en el arranque)
	Conflicts=no podrán estar activas al mismo tiempo
	Before=units que tienen que estar activas antes de ella
	After=units que se tienen que activar después de ella
	OnFailure=unidades que se activan cuando ésta entra en fail

[Service]

	Type= simple 
                    default si no tenemos BusName= y tenemos  ExecStart=(será el ppal proceso del servicio) 
                     
             forking 
                    el proceso configurado en ExecStart= abrirá una llamada fork() como parte de su  start-up. El proceso padre termina antes de que el starup de systemd siga con el start-up. Recomendable usar PIDFile=/var/run/servicio.pid

             oneshot 
                   similar a simple, pero el proceso termina antes de que systemd termine el startup. Útil RemainAfterExit=. El default si no tenemos ni Type= ni ExecStart=

             dbus similar a simple pero necesita BusName= (systemd and its auxiliary daemons expose a number of APIs on D-Bus)

             notify similar a simple pero envía notificación cuanto ha terminado de activarse

             idle  similar a simple pero la ejecución del servicio se retrasa hasta que todos se han ejecutado (consolas)
     

	RemainAfterExit= yes|no default no

	ExecStart=/ruta/comando1 \;  (mas de un comando si es oneshoot)
                  /ruta/comando2 \;
                  -/ruta/comando3 \;
                  ...


	ExecStartPre=


	ExecStartPost=


	ExecStopPost= si alguno de los Exec precedentes falla, se ejecutan estos



	ExecReload=/bin/kill -HUP $MAINPID

	ExecStop= se ejecuta solo si los otros Exec se han ejecutado correctamente


	RestartSec=tiempo de espera antes del Restart=

	Restart=cuando se tiene que hacer restart del servicio (no por systemd)
        no, on-success, on-failure, on-abnormal, on-watchdog,
           on-abort, or always


	TimeoutStartSec= tiempo de espera para el start-up

	TimeoutStopSec=    tiempo de espera para el   apagado  
 
	FailureAction=que se hace si el servicio entra en "failed"     


[Install]
	WantedBy=
